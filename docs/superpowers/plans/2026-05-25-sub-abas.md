```python
@router.get("/menu", response_model=List[AbaResponse])
def obter_menu_dinamico(
    db: Session = Depends(get_db),
    current_user: TokenPayload = Depends(get_current_user),
):
    abas = db.query(Aba).filter(Aba.ativo).order_by(Aba.ordem).all()

    if current_user.role != "admin":
        accessible_ids = {
            row[0] for row in db.query(UsuarioModulo.modulo_id)
            .filter(UsuarioModulo.usuario_id == int(current_user.sub))
            .all()
        }
        # Remove modules the user can't access
        for aba in abas:
            aba.modulos = [m for m in (aba.modulos or []) if m.id in accessible_ids]

    def build_tree(abas_list, parent_id=None):
        result = []
        for aba in abas_list:
            if aba.parent_id == parent_id:
                item = aba
                item.children = build_tree(abas_list, aba.id)
                # Prune empty branches for non-admin
                if current_user.role != "admin":
                    if not item.modulos and not item.children:
                        continue
                result.append(item)
        return result

    tree = build_tree(abas)
    return tree
```

- [ ] **Step 2: Update PUT /abas/{id} to accept optional parent_id**

```python
@router.put("/abas/{aba_id}", response_model=AbaResponse)
def atualizar_aba(
    aba_id: int,
    nome: str,
    icone: str,
    ordem: int,
    parent_id: int | None = None,
    db: Session = Depends(get_db),
    _: None = Depends(require_role("admin")),
):
    aba = db.query(Aba).filter(Aba.id == aba_id).first()
    if not aba:
        raise HTTPException(404, "Aba não encontrada")
    # Prevent self-reference cycle
    if parent_id is not None and parent_id == aba_id:
        raise HTTPException(422, "Uma aba não pode ser sub-aba de si mesma")
    aba.nome = nome
    aba.icone = icone
    aba.ordem = ordem
    aba.parent_id = parent_id
    db.commit()
    db.refresh(aba)
    return aba
```

- [ ] **Step 7: Update upsertAba and upsertModulo for recursive data**

```javascript
upsertAba(aba) {
    if (!aba?.id) return;
    // Find and update in tree, or append to root
    const updateInTree = (list) => {
        for (let i = 0; i < list.length; i++) {
            if (String(list[i].id) === String(aba.id)) {
                list[i] = { ...aba, modulos: list[i].modulos || [], children: list[i].children || [] };
                return true;
            }
            if (list[i].children && updateInTree(list[i].children)) return true;
        }
        return false;
    };
    if (!updateInTree(this.data)) {
        this.data.push({ ...aba, modulos: [], children: [] });
    }
    this.data.sort((a, b) => a.ordem - b.ordem);
}

upsertModulo(module, fallbackAbaId) {
    if (!module?.id) return;
    const targetAbaId = module.aba_id || fallbackAbaId;
    // Remove from all lists
    const removeFromTree = (list) => {
        for (const item of list) {
            item.modulos = item.modulos.filter(m => String(m.id) !== String(module.id));
            if (item.children) removeFromTree(item.children);
        }
    };
    removeFromTree(this.data);
    // Add to target
    const findAndAdd = (list) => {
        for (const item of list) {
            if (String(item.id) === String(targetAbaId)) {
                item.modulos.push(module);
                return true;
            }
            if (item.children && findAndAdd(item.children)) return true;
        }
        return false;
    };
    findAndAdd(this.data);
}
```
